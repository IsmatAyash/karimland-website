type Product
  @model
  @auth(rules: [{ allow: owner, queries: null }])
  @key(
    name: "queryName"
    fields: ["queryName", "ratings"]
    queryField: "productsByRating"
  ) {
  id: ID!
  title: String!
  images: [AWSURL!]!
  quantity: Int
  oldPrice: Float
  orders: Int
  avgRating: Float
  ratings: Int
  prodType: String!
  featured: Boolean!
  prices: [AWSJSON!]!
  description: [AWSJSON!]
  queryName: String!
}
